add -> track

push -> send
fetch -> check
pull -> update

branch -> fork
stash -> hold
pop -> restore
reset -> undo
log -> history
diff -> changes
status -> state
remote -> far
tag -> mark
bisect -> findbug
cherry-pick -> copycommit
